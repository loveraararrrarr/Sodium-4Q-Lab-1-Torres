const QUERY = window.location.search; const PARAMS = new URLSearchParams(QUERY); function processData() { let name = PARAMS.get("name"); document.getElementById("greatThanks").innerText = `Hello, ${name}!`; let amountAnak = PARAMS.get("amountAnak"); let amountAsawa = PARAMS.get("amountAsawa"); let payment = PARAMS.get("payment"); let costAnak = (amountAnak * 100); document.getElementById("costAnak").innerText = `anak: Php${costAnak}`; let costAsawa = (amountAsawa * 10); document.getElementById("costAsawa").innerText = `asawa: Php${costAsawa}`; let total = (costAnak + costAsawa); document.getElementById("total").innerText =`Total Cost: Php${total}`; if(payment < total) { let need = (total - payment); alert(`your money is kulang u still have to pay Php${need} more`); } else { let change = (payment - total); document.getElementById("change").innerText = `Change: Php${change} `; } }
